Task: This function should return n!
Below is my solution of the problem:

function factorial (n) {
    if (n < 0)
    return null;
                            
    if (n === 0 || n === 1)
    return 1;
                            
    for (let i = n; i > 1; i--) {
        n = n * (i - 1);
    };
    return n;
}
